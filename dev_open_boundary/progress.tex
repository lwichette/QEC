\begin{algorithm}

$iter\_sum_{0, k} = 0;$ \qquad \# for~ $(0, 0)$ ~and~ $(2\pi/L, 0)$\\
normalization\_factor = 0\;
\For{every lattice size L}{
    \For{every disorder e}{
        initialize interaction configuration\;
        initialize spin configuration\;
        \For{warm up period}{
            update lattice\;
          }
        \For{sample period}{
            update lattice\;
            iter\_sum += magnetic susceptibility for current spin config\;
            normalization\_factor $+= 1$\;
          }
      }
  }

$\zeta$ calculation from iter\_sums;\

\end{algorithm}
